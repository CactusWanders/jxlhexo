## float数据加减乘除
```
    var arr = [1,2,3]
    function accAdd(v1, v2) {
        <!-- 获取小数最大位数 -->
        let v1Len = v1.toString().split('.')[1] ? v1.toString().split('.')[1].length : 0;
        let v2Len = v2.toString().split('.')[1] ? v2.toString().split('.')[1].length : 0;
        let len = Math.max(v1Len,  v2Len);
        
        return (v1 * Math.pow(10, len) + v2 * Math.pow(10, len)) / Math.pow(10, len);
    };
    var res = arr.reduce(accAdd);
```
